scripted_gui = {
	allocation_decision_category_scripted_gui = {
		context_type = decision_category

		window_name = "allocation_decision_category_scripted_gui"

		effects = {
			entry_button_click = {
				ROOT = {
					if = {
						limit = {
							check_variable = { geck_target_state = PREV }
						}
						clear_variable = geck_target_state
					}
					else = {
						set_variable = { geck_target_state = PREV }
					}
				}
			}
			entry_button_right_click = {
				ROOT = {
					goto_state = PREV
				}
			}
		}

		triggers = {
			entry_button_click_enabled = {
				ROOT = {
					check_variable = { geck_in_use = 0 }
					PREV = {
						can_use_geck_in_state = yes
					}
				}
			}
		}

		dynamic_lists = {
			geck_decision_category_scripted_gui_gridbox = {
				array = owned_states
				entry_container = geck_decision_category_scripted_gui_entry
				change_scope = yes
			}
		}
	}
}
